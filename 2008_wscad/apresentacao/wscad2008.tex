\subsection{Benchmarks}

\begin{frame}
    \frametitle{Benchmarks e Dados para Testes (1/2)}
        \begin{itemize}
            \item Netperf - análise de rede
            \begin{itemize}
                \item Transferência de mensagens
                \item Protocolos TCP e UDP
                \item Grupos de 512 bytes, 512 Kbytes e 45 Mbytes
                \item Resultado:  Taxa de Trasnferência (X)bytes/s
            \end{itemize}
            \item dd - análise de disco
            \begin{itemize}
                \item Transferência de arquivos
                \item Tamanhos: 128Mbytes à 4Gbytes
                \item Bloco padrão de 512 bytes
                \item Resultado: Taxa de Transferência(Mbytes/s) e Tempo(s)
            \end{itemize}
        \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Benchmarks e Dados para Testes (2/2)}
        \begin{itemize}
            \item Stream - análise de memória
            \begin{itemize}
                \item Largura de banda
                \item Resultado em Mbytes/s
                \item Copy      a(i) = b(i)
                \item Scale a(i) = q*b(i)
                \item Add       a(i) = b(i)+c(i)
                \item Triad a(i) = b(i)+q*c(i)
            \end{itemize}
            \item Linpack-PC - análise de CPU
            \begin{itemize}
                \item Conjunto de rotinas 
                \item Matrizes 100 x 100
                \item Precisão simples
                \item Resultado: Número de operações em ponto flutuante (Mflops)
            \end{itemize}
        \end{itemize}
\end{frame}
